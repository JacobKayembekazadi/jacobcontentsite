  <!-- Minimal JS for nav highlight, smooth scroll, reduced motion -->
  <script>
    // Smooth scroll for anchor links
    document.querySelectorAll('a[href^="#"]').forEach(link => {
      link.addEventListener('click', e => {
        const target = document.querySelector(link.getAttribute('href'));
        if (target) {
          e.preventDefault();
          target.scrollIntoView({ behavior: window.matchMedia('(prefers-reduced-motion: reduce)').matches ? 'auto' : 'smooth' });
        }
      });
    });

    // IntersectionObserver for nav highlight
    const navLinks = document.querySelectorAll('[data-nav]');
    const sections = Array.from(navLinks).map(link => document.querySelector(link.getAttribute('href')));
    const navActive = (id) => {
      navLinks.forEach(link => link.classList.toggle('nav__link--active', link.getAttribute('href') === '#' + id));
    };
    const observer = new IntersectionObserver(entries => {
      entries.forEach(entry => {
        if (entry.isIntersecting) navActive(entry.target.id);
      });
    }, { threshold: 0.5 });
    sections.forEach(section => section && observer.observe(section));

    // Calendly modal (placeholder)
    document.querySelectorAll('[data-calendly]').forEach(btn => {
      btn.addEventListener('click', e => {
        e.preventDefault();
        document.getElementById('calendly').scrollIntoView({ behavior: window.matchMedia('(prefers-reduced-motion: reduce)').matches ? 'auto' : 'smooth' });
      });
    });
  </script>
